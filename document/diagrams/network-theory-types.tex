\begin{grammar}	
	<elementType> $\ni \elementTypeVar$ ::= \missing
	\grammarShrink

	<shape> $\ni \shapeVar$ ::= [$n$,...,$n$] \hspace{5em} where $n \in \mathbb{N}$ 
	\grammarShrink

	<tensorType> $\ni \tensorTypeVar$ ::= <elementType>  $\times$ <shape>
	\grammarShrink
	
	<modelType> $\ni \modelTypeVar$ ::= <tensorType>$^+$ $\times$ <tensor-type>$^+$
\end{grammar}
